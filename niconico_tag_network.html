adjacency <- function(n){
## 
mydata <- diag(0,length(colnames(n)))
## 
colnames(mydata) <- colnames(n)
rownames(mydata) <- colnames(n)
## r
for(r in 1:length(colnames(n))){
## r
mydata[r,] <- colSums(
## r1
n[n[,r] == 1,]
)
}
mydata
}
